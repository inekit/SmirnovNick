RETURN NUMBER
AS
c NUMBER;
BEGIN
c:=ca;

LOOP

elsif c mod 2 = 1 then c:=c*3+1;
end if;
dbms_output.put_line(c); 

END LOOP;
END;

AS

 select a_id,count(k_id) k,name
 from k_a,animals a where a.id=k_a.a_id
